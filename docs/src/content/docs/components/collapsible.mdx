## Usage

<Tabs>
	<TabItem label="Builder">
```svelte
<script lang="ts">
	import { Collapsible } from "melt";

	const collapsible = new Collapsible();
</script>

<div {...collapsible.root}>
    <button {...collapsible.trigger}>Open/Close content</button>

    {#if collapsible.open}
        <div {...collapsible.content}>Content</div>
    {/if}
</div>
```
	</TabItem>

	<TabItem label="Component">
```svelte
<script lang="ts">
	import { Collapsible } from "melt/components";
</script>

<Collapsible>
    {#snippet children(collapsible)}
        <div {...collapsible.root}>
            <button {...collapsible.trigger}>Open/Close content</button>
            
            {#if collapsible.open}
                <div {...collapsible.content}>Content</div>
            {/if}
        </div>
    {/snippet}
</Collapsible>
```
	</TabItem>
